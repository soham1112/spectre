29 /// \brief ParallelComponent representing a set of points to be interpolated
30 /// to and a function to call upon interpolation to those points.
31 ///
32 /// Each InterpolationTarget will communicate with the `Interpolator`.
33 ///
34 /// `InterpolationTargetTag` must contain the following type aliases:
35 /// - vars_to_interpolate_to_target:
36 /// A `tmpl::list` of tags describing variables to interpolate.
37 /// Will be used to construct a `Variables`.
38 /// - compute_items_on_source:
39 /// A `tmpl::list` of compute items that uses
40 /// `Metavariables::interpolator_source_vars` as input and computes the
41 /// `Variables` defined by `vars_to_interpolate_to_target`.
42 /// - compute_items_on_target:
43 /// A `tmpl::list` of compute items that uses
44 /// `vars_to_interpolate_to_target` as input.
45 /// - compute_target_points:
46 /// A `simple_action` of `InterpolationTarget`
47 /// that computes the target points and sends them to `Interpolators`. It
48 /// takes a `temporal_id` as an extra argument. `compute_target_points`
49 /// can (optionally) have an additional function
50 ///```
51 /// static auto initialize(db::DataBox<DbTags>&&,
52 /// const Parallel::ConstGlobalCache<Metavariables>&)
53 /// noexcept;
54 ///```
55 /// that adds arbitrary tags to the `DataBox` when the
56 /// `InterpolationTarget` is initialized. If `compute_target_points` has
57 /// an `initialize` function, it must also have a type alias
58 /// `initialization_tags` which is a `tmpl::list` of the tags that are
59 /// added by `initialize`.
60 /// - post_interpolation_callback:
61 /// A struct with a type alias `const_global_cache_tags` (listing tags that
62 /// should be read from option parsing), with a type alias
63 /// `observation_types` (listing any ObservationTypes that the callback
64 /// will use in constructing ObserverIds to call
65 /// observers::ThreadedActions::WriteReductionData), and with a function
66 ///```
67 /// void apply(const DataBox<DbTags>&,
68 /// const intrp::ConstGlobalCache<Metavariables>&,
69 /// const Metavariables::temporal_id&) noexcept;
70 ///```
71 /// or
72 ///```
73 /// bool apply(const gsl::not_null<db::DataBox<DbTags>*>,
74 /// const gsl::not_null<intrp::ConstGlobalCache<Metavariables>*>,
75 /// const Metavariables::temporal_id&) noexcept;
76 ///```
77 /// that will be called when interpolation is complete. `DbTags` includes
78 /// everything in `vars_to_interpolate_to_target`, plus everything in
79 /// `compute_items_on_target`. The second form of the `apply` function
80 /// should return false only if it calls another `intrp::Action` that still
81 /// needs the volume data at this temporal_id (such as another iteration of
82 /// the horizon finder).
83 ///
84 /// `Metavariables` must contain the following type aliases:
85 /// - interpolator_source_vars:
86 /// A `tmpl::list` of tags that define a `Variables` sent from all
87 /// `Element`s to the local `Interpolator`.
88 /// - interpolation_target_tags:
89 /// A `tmpl::list` of all `InterpolationTargetTag`s.
90 /// - temporal_id:
91 /// The type held by ::intrp::Tags::TemporalIds.
92 /// - domain_frame:
93 /// The `::Frame` of the Domain.
94 ///
95 /// `Metavariables` must contain the following static constexpr members:
96 /// - size_t domain_dim:
97 /// The dimension of the Domain.
98 template <class Metavariables, typename InterpolationTargetTag>
99 struct InterpolationTarget {
100  using chare_type = ::Parallel::Algorithms::Singleton;
101  using metavariables = Metavariables;
102  using action_list = tmpl::list<>;
103  using initial_databox =
104  db::compute_databox_type<typename Actions::InitializeInterpolationTarget<
105  InterpolationTargetTag>::template return_tag_list<Metavariables>>;
106  using options = tmpl::list<::OptionTags::DomainCreator<
107  Metavariables::domain_dim, typename Metavariables::domain_frame>>;
108  using const_global_cache_tag_list = Parallel::get_const_global_cache_tags<
109  tmpl::list<typename InterpolationTargetTag::compute_target_points,
110  typename InterpolationTargetTag::post_interpolation_callback>>;
111 
112  static void initialize(
113  Parallel::CProxy_ConstGlobalCache<metavariables>& global_cache,
114  std::unique_ptr<DomainCreator<Metavariables::domain_dim,
115  typename Metavariables::domain_frame>>
116  domain_creator) noexcept;
117  static void execute_next_phase(
118  typename metavariables::Phase next_phase,
119  Parallel::CProxy_ConstGlobalCache<metavariables>& global_cache) noexcept {
120  try_register_with_observers(next_phase, global_cache);
121  }
122 
123  private:
124  template <typename PhaseType,
125  Requires<not observers::has_register_with_observer_v<PhaseType>> =
126  nullptr>
127  static void try_register_with_observers(
128  const PhaseType /*next_phase*/,
129  Parallel::CProxy_ConstGlobalCache<
130  Metavariables>& /*global_cache*/) noexcept {}
131  template <
132  typename PhaseType,
133  Requires<observers::has_register_with_observer_v<PhaseType>> = nullptr>
134  static void try_register_with_observers(
135  const PhaseType next_phase,
136  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache) noexcept {
137  if (next_phase == Metavariables::Phase::RegisterWithObserver) {
138  auto& local_cache = *(global_cache.ckLocalBranch());
139  tmpl::for_each<
140  typename InterpolationTargetTag::post_interpolation_callback::
141  observation_types>([&local_cache](auto type_v) noexcept {
142  using type = typename decltype(type_v)::type;
143  // The 'time' value of the observation_id doesn't matter here and
144  // is currently not used.
145  // In the future when we do load balancing, we will need to register
146  // and unregister at specific times.
147  const observers::ObservationId observation_id(0., type{});
148  Parallel::simple_action<
149  observers::Actions::RegisterSingletonWithObserverWriter>(
150  Parallel::get_parallel_component<
151  InterpolationTarget<Metavariables, InterpolationTargetTag>>(
152  local_cache),
153  observation_id);
154  });
155  }
156  }
157 };

160 void InterpolationTarget<Metavariables, InterpolationTargetTag>::initialize(

169  my_proxy, domain_creator->create_domain());
