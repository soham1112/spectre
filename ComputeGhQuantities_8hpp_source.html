170 /*!
171  * \ingroup GeneralRelativityGroup
172  * \brief Computes time derivative of the spatial metric.
173  *
174  * \details Let the generalized harmonic conjugate momentum and spatial
175  * derivative variables be \f$\Pi_{ab} = -t^c \partial_c \psi_{ab} \f$ and
176  * \f$\Phi_{iab} = \partial_i \psi_{ab} \f$. As \f$ t_i \equiv 0 \f$. The time
177  * derivative of the spatial metric is given by the time derivative of the
178  * spatial sector of the spacetime metric, i.e.
179  * \f$ \partial_0 g_{ij} = \partial_0 \psi_{ij} \f$.
180  *
181  * To compute the latter, we use the evolution equation for \f$ \psi_{ij} \f$,
182  * c.f. eq.(35) of \cite Lindblom2005qh (with \f$\gamma_1 = -1\f$):
183  *
184  * \f[
185  * \partial_0 \psi_{ab} = - N \Pi_{ab} + N^k \Phi_{kab}
186  * \f]
187  */
188 template <size_t SpatialDim, typename Frame, typename DataType>
189 void time_deriv_of_spatial_metric(
190  gsl::not_null<tnsr::ii<DataType, SpatialDim, Frame>*> dt_spatial_metric,
191  const Scalar<DataType>& lapse,
192  const tnsr::I<DataType, SpatialDim, Frame>& shift,
193  const tnsr::iaa<DataType, SpatialDim, Frame>& phi,
194  const tnsr::aa<DataType, SpatialDim, Frame>& pi) noexcept;

230 /*!
231  * \ingroup GeneralRelativityGroup
232  * \brief Computes spatial derivatives of lapse (N) from the generalized
233  * harmonic variables and spacetime unit normal 1-form.
234  *
235  * \details If the generalized harmonic conjugate momentum and spatial
236  * derivative variables are \f$\Pi_{ab} = -t^c \partial_c \psi_{ab} \f$ and
237  * \f$\Phi_{iab} = \partial_i \psi_{ab} \f$, the spatial derivatives of N
238  * can be obtained from:
239  * \f{align*}
240  * t^a t^b \Phi_{iab} = -\frac{1}{2N} [\partial_i (-N^2 + N_jN^j)-
241  * 2 N^j \partial_i N_j
242  * + N^j N^k \partial_i g_{jk}]
243  * = -\frac{2}{N} \partial_i N,
244  * \f}
245  * since
246  * \f[
247  * \partial_i (N_jN^j) = 2N^j \partial_i N_j - N^j N^k \partial_i g_{jk}.
248  * \f]
249  *
250  * \f[
251  * \Longrightarrow \partial_i N = -(N/2) t^a \Phi_{iab} t^b
252  * \f]
253  */

255 void spatial_deriv_of_lapse(
256  gsl::not_null<tnsr::i<DataType, SpatialDim, Frame>*> deriv_lapse,
257  const Scalar<DataType>& lapse,
258  const tnsr::A<DataType, SpatialDim, Frame>& spacetime_unit_normal,
259  const tnsr::iaa<DataType, SpatialDim, Frame>& phi) noexcept;
260 
261 template <size_t SpatialDim, typename Frame, typename DataType>

269 /*!
270  * \ingroup GeneralRelativityGroup
271  * \brief Computes time derivative of lapse (N) from the generalized
272  * harmonic variables, lapse, shift and the spacetime unit normal 1-form.
273  *
274  * \details Let the generalized harmonic conjugate momentum and spatial
275  * derivative variables be \f$\Pi_{ab} = -t^c \partial_c \psi_{ab} \f$ and
276  * \f$\Phi_{iab} = \partial_i \psi_{ab} \f$, and the operator
277  * \f$D := \partial_0 - N^k \partial_k \f$. The time derivative of N is then:
278  * \f{align*}
279  * \frac{1}{2} N^2 t^a t^b \Pi_{ab} - \frac{1}{2} N N^i t^a t^b \Phi_{iab}
280  * =& \frac{1}{2} N^2 t^a t^b t^c \partial_c \psi_{ab}
281  * - \frac{1}{2} N N^i (-(2/N) \partial_i N) \\
282  * =& \frac{1}{2} N^2 [-(1/N^3) D[g_{jk} N^j N^k - N^2] \\
283  * &- (N^j N^k / N^3)D[g_{jk}] \\
284  * &+ 2 (N^j / N^3) D[g_{jk} N^k] + (2 / N^2)(N^i \partial_i N)] \\
285  * =& \frac{1}{2N} [-D[g_{jk}N^jN^k - N^2] - N^jN^k D[g_{jk}]
286  * + 2N N^k\partial_k N + 2N^j D[g_{jk}N^k]] \\
287  * =& D[N] + N^k\partial_k N \\
288  * =& \partial_0 N
289  * \f}
290  * where the simplification done for \f$\partial_i N\f$ is used to substitute
291  * for the second term (\f$\frac{1}{2} N N^i t^a t^b \Phi_{iab}\f$).
292  *
293  * Thus,
294  * \f[
295  * \partial_0 N = (N/2)(N t^a t^b \Pi_{ab} - N^i t^a t^b \Phi_{iab})
296  * \f]
297  */
298 template <size_t SpatialDim, typename Frame, typename DataType>
299 void time_deriv_of_lapse(
300  gsl::not_null<Scalar<DataType>*> dt_lapse, const Scalar<DataType>& lapse,
301  const tnsr::I<DataType, SpatialDim, Frame>& shift,
302  const tnsr::A<DataType, SpatialDim, Frame>& spacetime_unit_normal,
303  const tnsr::iaa<DataType, SpatialDim, Frame>& phi,
304  const tnsr::aa<DataType, SpatialDim, Frame>& pi) noexcept;
305 
306 template <size_t SpatialDim, typename Frame, typename DataType>
307 Scalar<DataType> time_deriv_of_lapse(
308  const Scalar<DataType>& lapse,
309  const tnsr::I<DataType, SpatialDim, Frame>& shift,

316 /*!
317  * \ingroup GeneralRelativityGroup
318  * \brief Computes spatial derivatives of the shift vector from
319  * the generalized harmonic and geometric variables
320  *
321  * \details Spatial derivatives of the shift vector \f$N^i\f$ can be derived
322  * from the following steps:
323  * \f{align*}
324  * \partial_i N^j
325  * =& g^{jl} g_{kl} \partial_i N^k \\
326  * =& g^{jl} (N^k \partial_i g_{lk}
327  * + g_{kl}\partial_i N^k - N^k \partial_i g_{kl}) \\
328  * =& g^{jl} (\partial_i N_l - N^k \partial_i g_{lk}) (\because g^{j0} = 0) \\
329  * =& g^{ja} (\partial_i \psi_{a0} - N^k \partial _i \psi_{ak}) \\
330  * =& N g^{ja} t^b \partial_i \psi_{ab} \\
331  * =& (g^{ja} - t^j t^a) N t^b \Phi_{iab} - 2 t^j \partial_i N \\
332  * =& \psi^{ja} N t^b \Phi_{iab} - 2 t^j \partial_i N \\
333  * =& N (\psi^{ja} + t^j t^a) t^b \Phi_{iab}.
334  * \f}
335  * where we used the equation from spatial_deriv_of_lapse() for
336  * \f$\partial_i N\f$.
337  */
338 template <size_t SpatialDim, typename Frame, typename DataType>
339 void spatial_deriv_of_shift(
340  gsl::not_null<tnsr::iJ<DataType, SpatialDim, Frame>*> deriv_shift,
341  const Scalar<DataType>& lapse,
342  const tnsr::AA<DataType, SpatialDim, Frame>& inverse_spacetime_metric,
343  const tnsr::A<DataType, SpatialDim, Frame>& spacetime_unit_normal,
344  const tnsr::iaa<DataType, SpatialDim, Frame>& phi) noexcept;

355 /*!
356  * \ingroup GeneralRelativityGroup
357  * \brief Computes time derivative of the shift vector from
358  * the generalized harmonic and geometric variables
359  *
360  * \details The time derivative of \f$ N^i \f$ can be derived from the following
361  * steps:
362  * \f{align*}
363  * \partial_0 N^i
364  * =& g^{ik} \partial_0 (g_{kj} N^j) - N^j g^{ik} \partial_0 g_{kj} \\
365  * =& N g^{ik} t^b \partial_0 \psi_{kb} \\
366  * =& N g^{ik} t^b (\partial_0 - N^j\partial_j) \psi_{kb}
367  * + N g^{ik} t^b N^j\partial_j \psi_{kb} \\
368  * =& -N^2 t^b\Pi_{kb} g^{ik} + N N^j t^b\Phi_{jkb} g^{ik} \\
369  * =& -N g^{ik} t^b (N \Pi_{kb} - N^j \Phi_{jkb}) \\
370  * \f}
371  */

373 void time_deriv_of_shift(
