144  enum class Phase {
145  Initialization,
146  RegisterWithObserver,
147  Evolve,
148  Exit
149  };
150 
151  static Phase determine_next_phase(
152  const Phase& current_phase,
153  const Parallel::CProxy_ConstGlobalCache<
154  EvolutionMetavars>& /*cache_proxy*/) noexcept {
155  switch (current_phase) {
156  case Phase::Initialization:
157  return Phase::RegisterWithObserver;
158  case Phase::RegisterWithObserver:
159  return Phase::Evolve;
160  case Phase::Evolve:
161  return Phase::Exit;
162  case Phase::Exit:
163  ERROR(
164  "Should never call determine_next_phase with the current phase "
165  "being 'Exit'");
166  default:
167  ERROR(
168  "Unknown type of phase. Did you static_cast<Phase> an integral "
169  "value?");
170  }
171  }
