34 /// \brief Initialize items related to the basic structure of the Domain
35 ///
36 /// DataBox changes:
37 /// - Adds:
38 /// * `Tags::Mesh<Dim>`
39 /// * `Tags::Element<Dim>`
40 /// * `Tags::ElementMap<Dim>`
41 /// * `Tags::LogicalCoordinates<Dim>`
42 /// * `Tags::MappedCoordinates`
43 /// * `Tags::InverseJacobian`
44 /// * `Tags::MinimumGridSpacing<Dim, Frame::Inertial>>`
45 /// - Removes: nothing
46 /// - Modifies: nothing
47 template <size_t Dim>
48 struct Domain {
49  using simple_tags = db::AddSimpleTags<Tags::Mesh<Dim>, Tags::Element<Dim>,
50  Tags::ElementMap<Dim>>;
51 
52  using compute_tags =
53  db::AddComputeTags<Tags::LogicalCoordinates<Dim>,
54  Tags::MappedCoordinates<Tags::ElementMap<Dim>,
55  Tags::LogicalCoordinates<Dim>>,
56  Tags::InverseJacobian<Tags::ElementMap<Dim>,
57  Tags::LogicalCoordinates<Dim>>,
58  Tags::MinimumGridSpacing<Dim, Frame::Inertial>>;
59 
60  template <typename TagsList>
61  static auto initialize(
62  db::DataBox<TagsList>&& box, const ElementIndex<Dim>& array_index,
63  const std::vector<std::array<size_t, Dim>>& initial_extents,
64  const ::Domain<Dim, Frame::Inertial>& domain) noexcept {
65  const ElementId<Dim> element_id{array_index};
66  const auto& my_block = domain.blocks()[element_id.block_id()];
67  Mesh<Dim> mesh = element_mesh(initial_extents, element_id);
68  Element<Dim> element = create_initial_element(element_id, my_block);
69  ElementMap<Dim, Frame::Inertial> map{element_id,
70  my_block.coordinate_map().get_clone()};
71 
72  return db::create_from<db::RemoveTags<>, simple_tags, compute_tags>(
73  std::move(box), std::move(mesh), std::move(element), std::move(map));
74  }
75 };
76
