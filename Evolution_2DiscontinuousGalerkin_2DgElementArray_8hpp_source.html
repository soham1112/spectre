75  static void execute_next_phase(
76  const typename Metavariables::Phase next_phase,
77  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache) noexcept {
78  auto& local_cache = *(global_cache.ckLocalBranch());
79  if (next_phase == Metavariables::Phase::Evolve) {
80  Parallel::get_parallel_component<DgElementArray>(local_cache)

96  template <
97  typename PhaseType,
98  Requires<observers::has_register_with_observer_v<PhaseType>> = nullptr>
99  static void try_register_with_observers(
100  const PhaseType next_phase,
101  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache) noexcept {
102  if (next_phase == Metavariables::Phase::RegisterWithObserver) {
103  auto& local_cache = *(global_cache.ckLocalBranch());
104  // We currently use an observation_id with a fake time when registering
105  // observers but in the future when we start doing load balancing and
106  // elements migrate around the system they will need to register and
107  // unregister themselves at specific times.
108  const observers::ObservationId observation_id_with_fake_time(
109  0., typename Metavariables::element_observation_type{});
110  Parallel::simple_action<observers::Actions::RegisterWithObservers<
111  observers::TypeOfObservation::ReductionAndVolume>>(
112  Parallel::get_parallel_component<DgElementArray>(local_cache),
113  observation_id_with_fake_time);
114  }
115  }

127 template <class Metavariables, class InitializeAction, class ActionList>
128 void DgElementArray<Metavariables, InitializeAction, ActionList>::initialize(
129  Parallel::CProxy_ConstGlobalCache<Metavariables>& global_cache,
130  const std::unique_ptr<DomainCreator<volume_dim, Frame::Inertial>>
131  domain_creator,
132  const double initial_time, const double initial_dt,
133  const double initial_slab_size) noexcept {
134  auto& cache = *global_cache.ckLocalBranch();
135  auto& dg_element_array =
136  Parallel::get_parallel_component<DgElementArray>(cache);
137 
138  if (not Metavariables::local_time_stepping and
139  std::abs(initial_dt) != initial_slab_size) {
140  ERROR("Step and slab size must agree for global time-stepping.");
141  }
142 
143  auto domain = domain_creator->create_domain();
144  for (const auto& block : domain.blocks()) {
145  const auto initial_ref_levs =
146  domain_creator->initial_refinement_levels()[block.id()];
147  const std::vector<ElementId<volume_dim>> element_ids =
148  initial_element_ids(block.id(), initial_ref_levs);
149  int which_proc = 0;
150  const int number_of_procs = Parallel::number_of_procs();
151  for (size_t i = 0; i < element_ids.size(); ++i) {
152  dg_element_array(ElementIndex<volume_dim>(element_ids[i]))
153  .insert(global_cache, which_proc);
154  which_proc = which_proc + 1 == number_of_procs ? 0 : which_proc + 1;
155  }
156  }
157  dg_element_array.doneInserting();
158 
159  Parallel::simple_action<InitializeAction>(
160  dg_element_array, domain_creator->initial_extents(), std::move(domain),
161  initial_time, initial_dt, initial_slab_size);
162 }
