66 /// \ingroup ActionsGroup
67 /// \ingroup DiscontinuousGalerkinGroup
68 /// \brief Initialize a dG element with analytic initial data
69 ///
70 /// Uses:
71 /// - ConstGlobalCache:
72 /// * A tag deriving off of Cache::AnalyticSolutionBase
73 /// * OptionTags::TimeStepper
74 ///
75 /// DataBox changes:
76 /// - Adds:
77 /// * Tags::TimeId
78 /// * Tags::Time
79 /// * Tags::TimeStep
80 /// * Tags::LogicalCoordinates<Dim>
81 /// * Tags::Mesh<Dim>
82 /// * Tags::Element<Dim>
83 /// * Tags::ElementMap<Dim>
84 /// * System::variables_tag
85 /// * Tags::HistoryEvolvedVariables<System::variables_tag,
86 /// db::add_tag_prefix<Tags::dt, System::variables_tag>>
87 /// * Tags::Coordinates<Tags::ElementMap<Dim>,
88 /// Tags::LogicalCoordinates<Dim>>
89 /// * Tags::InverseJacobian<Tags::ElementMap<Dim>,
90 /// Tags::LogicalCoordinates<Dim>>
91 /// * Tags::deriv<System::gradients_tags>
92 /// * db::add_tag_prefix<Tags::dt, System::variables_tag>
93 /// * Tags::UnnormalizedFaceNormal<Dim>
94 /// - Removes: nothing
95 /// - Modifies: nothing
96 template <size_t Dim>
97 struct InitializeElement {
98  static Mesh<Dim> element_mesh(
99  const std::vector<std::array<size_t, Dim>>& initial_extents,
100  const ElementId<Dim>& element_id,
101  const OrientationMap<Dim>& orientation = {}) noexcept {
102  const auto& unoriented_extents = initial_extents[element_id.block_id()];
103  Index<Dim> extents;
104  for (size_t i = 0; i < Dim; ++i) {
105  extents[i] = gsl::at(unoriented_extents, orientation(i));
106  }
107  return {extents.indices(), Spectral::Basis::Legendre,
108  Spectral::Quadrature::GaussLobatto};
109  }

311  // Global time stepping
312  template <typename Metavariables,
313  Requires<not Metavariables::local_time_stepping> = nullptr>

315  const Time& initial_time, const double initial_dt_value,
316  const Parallel::ConstGlobalCache<Metavariables>& /*cache*/) noexcept {
317  return (initial_dt_value > 0.0 ? 1 : -1) * initial_time.slab().duration();

325  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
326  const auto& step_controller =
327  Parallel::get<OptionTags::StepController>(cache);

331  template <typename TagsList, typename Metavariables>
332  static auto initialize(
333  db::DataBox<TagsList>&& box,
334  const Parallel::ConstGlobalCache<Metavariables>& cache,
335  const double initial_time_value, const double initial_dt_value,
336  const double initial_slab_size) noexcept {
337  using DtVars = typename dt_variables_tag::type;
338 
339  const size_t num_grid_points =
340  db::get<Tags::Mesh<Dim>>(box).number_of_grid_points();
341 
342  // Will be overwritten before use
343  DtVars dt_vars{num_grid_points};
344 
345  const bool time_runs_forward = initial_dt_value > 0.0;
346  const Slab initial_slab =
347  time_runs_forward ? Slab::with_duration_from_start(initial_time_value,
348  initial_slab_size)
349  : Slab::with_duration_to_end(initial_time_value,
350  initial_slab_size);
351  const Time initial_time =
352  time_runs_forward ? initial_slab.start() : initial_slab.end();
353  const TimeDelta initial_dt =
354  get_initial_time_step(initial_time, initial_dt_value, cache);
355 
356  const auto& time_stepper = Parallel::get<OptionTags::TimeStepper>(cache);
357 
358  typename Tags::HistoryEvolvedVariables<variables_tag,
359  dt_variables_tag>::type history;
360  // This is stored as Tags::Next<Tags::TimeId> and will be used
361  // to update Tags::TimeId at the start of the algorithm.
362  //
363  // The slab number is increased in the self-start phase each
364  // time one order of accuracy is obtained, and the evolution
365  // proper starts with slab 0.
366  const TimeId time_id =
367  TimeId(time_runs_forward,
368  -static_cast<int64_t>(time_stepper.number_of_past_steps()),
369  initial_time);
370 
371  return db::create_from<db::RemoveTags<>, simple_tags, compute_tags>(
372  std::move(box), TimeId{}, time_id, initial_dt, std::move(dt_vars),
373  std::move(history));
374  }

416  template <typename TagsList>
417  static auto add_mortar_data(
418  db::DataBox<TagsList>&& box,
419  const std::vector<std::array<size_t, Dim>>& initial_extents) noexcept {
420  const auto& element = db::get<Tags::Element<Dim>>(box);
421  const auto& mesh = db::get<Tags::Mesh<Dim>>(box);
422 
423  typename mortar_data_tag::type mortar_data{};
424  typename Tags::Mortars<Tags::Next<temporal_id_tag>, Dim>::type
425  mortar_next_temporal_ids{};
426  typename Tags::Mortars<Tags::Mesh<Dim - 1>, Dim>::type mortar_meshes{};
427  typename Tags::Mortars<Tags::MortarSize<Dim - 1>, Dim>::type
428  mortar_sizes{};
429  const auto& temporal_id = get<Tags::Next<temporal_id_tag>>(box);
430  for (const auto& direction_neighbors : element.neighbors()) {
431  const auto& direction = direction_neighbors.first;
432  const auto& neighbors = direction_neighbors.second;
433  for (const auto& neighbor : neighbors) {
434  const auto mortar_id = std::make_pair(direction, neighbor);
435  mortar_data[mortar_id]; // Default initialize data
436  mortar_next_temporal_ids.insert({mortar_id, temporal_id});
437  mortar_meshes.emplace(
438  mortar_id,
439  dg::mortar_mesh(mesh.slice_away(direction.dimension()),
440  element_mesh(initial_extents, neighbor,
441  neighbors.orientation())
442  .slice_away(direction.dimension())));
443  mortar_sizes.emplace(
444  mortar_id,
445  dg::mortar_size(element.id(), neighbor, direction.dimension(),
446  neighbors.orientation()));
447  }
448  }
449 
450  for (const auto& direction : element.external_boundaries()) {
451  const auto mortar_id = std::make_pair(
452  direction, ElementId<Dim>::external_boundary_id());
453  mortar_data[mortar_id];
454  // Since no communication needs to happen for boundary conditions,
455  // the temporal id is not advanced on the boundary, so we set it equal
456  // to the current temporal id in the element
457  mortar_next_temporal_ids.insert({mortar_id, temporal_id});
458  mortar_meshes.emplace(mortar_id,
459  mesh.slice_away(direction.dimension()));
460  mortar_sizes.emplace(mortar_id,
461  make_array<Dim - 1>(Spectral::MortarSize::Full));
462  }
463 
464  return db::create_from<
465  db::RemoveTags<>,
466  db::AddSimpleTags<mortar_data_tag,
467  Tags::Mortars<Tags::Next<temporal_id_tag>, Dim>,
468  Tags::Mortars<Tags::Mesh<Dim - 1>, Dim>,
469  Tags::Mortars<Tags::MortarSize<Dim - 1>, Dim>>>(
470  std::move(box), std::move(mortar_data),
471  std::move(mortar_next_temporal_ids), std::move(mortar_meshes),
472  std::move(mortar_sizes));
473  }
