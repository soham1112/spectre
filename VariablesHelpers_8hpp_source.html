41 /*!
42  * \ingroup DataStructuresGroup
43  * \brief Slices the data within `volume_tensor` to a codimension 1 slice. The
44  * slice has a constant logical coordinate in direction `sliced_dim`,
45  * slicing the volume at `fixed_index` in that dimension. For
46  * example, to get the lower boundary of `sliced_dim`, pass `0` for
47  * `fixed_index`; to get the upper boundary, pass
48  * `extents[sliced_dim] - 1`.
49  *
50  * \see add_slice_to_data
51  *
52  * returns Tensor class sliced to a hypersurface.
53  */
54 template <std::size_t VolumeDim, typename VectorType, typename... Structure>
55 void data_on_slice(
56  const gsl::not_null<Tensor<VectorType, Structure...>*> interface_tensor,
57  const Tensor<VectorType, Structure...>& volume_tensor,
58  const Index<VolumeDim>& element_extents, const size_t sliced_dim,
59  const size_t fixed_index) noexcept {
60  const size_t interface_grid_points =
61  element_extents.slice_away(sliced_dim).product();
62  if (interface_tensor->begin()->size() != interface_grid_points) {
63  *interface_tensor = Tensor<VectorType, Structure...>(interface_grid_points);
64  }
65 
66  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
67  for (decltype(auto) interface_and_volume_tensor_components :
68  boost::combine(*interface_tensor, volume_tensor)) {
69  boost::get<0>(interface_and_volume_tensor_components)[si.slice_offset()] =
70  boost::get<1>(
71  interface_and_volume_tensor_components)[si.volume_offset()];
72  }
73  }
74 }

90 /*!
91  * \ingroup DataStructuresGroup
92  * \brief Slices the data within `vars` to a codimension 1 slice. The
93  * slice has a constant logical coordinate in direction `sliced_dim`,
94  * slicing the volume at `fixed_index` in that dimension. For
95  * example, to get the lower boundary of `sliced_dim`, pass `0` for
96  * `fixed_index`; to get the upper boundary, pass
97  * `extents[sliced_dim] - 1`.
98  *
99  * \see add_slice_to_data
100  *
101  * returns Variables class sliced to a hypersurface.
102  */
103 template <std::size_t VolumeDim, typename TagsList>
104 void data_on_slice(const gsl::not_null<Variables<TagsList>*> interface_vars,
105  const Variables<TagsList>& vars,
106  const Index<VolumeDim>& element_extents,
107  const size_t sliced_dim, const size_t fixed_index) noexcept {
108  const size_t interface_grid_points =
109  element_extents.slice_away(sliced_dim).product();
110  const size_t volume_grid_points = vars.number_of_grid_points();
111  constexpr const size_t number_of_independent_components =
112  Variables<TagsList>::number_of_independent_components;
113 
114  if (interface_vars->number_of_grid_points() != interface_grid_points) {
115  *interface_vars = Variables<TagsList>(interface_grid_points);
116  }
117  using value_type = typename Variables<TagsList>::value_type;
118  const value_type* vars_data = vars.data();
119  value_type* interface_vars_data = interface_vars->data();
120  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
121  for (size_t i = 0; i < number_of_independent_components; ++i) {
122  // clang-tidy: do not use pointer arithmetic
123  interface_vars_data[si.slice_offset() + // NOLINT
124  i * interface_grid_points] = // NOLINT
125  vars_data[si.volume_offset() + i * volume_grid_points]; // NOLINT
126  }
127  }
128 }

144 /*!
145  * \ingroup DataStructuresGroup
146  * \brief Slices volume `Tensor`s into a `Variables`
147  *
148  * The slice has a constant logical coordinate in direction `sliced_dim`,
149  * slicing the volume at `fixed_index` in that dimension. For
150  * example, to get the lower boundary of `sliced_dim`, pass `0` for
151  * `fixed_index`; to get the upper boundary, pass
152  * `extents[sliced_dim] - 1`.
153  */
154 template <typename... TagsToSlice, size_t VolumeDim>
155 void data_on_slice(
156  const gsl::not_null<Variables<tmpl::list<TagsToSlice...>>*> interface_vars,
157  const Index<VolumeDim>& element_extents, const size_t sliced_dim,
158  const size_t fixed_index,
159  const typename TagsToSlice::type&... tensors) noexcept {
160  const size_t interface_grid_points =
161  element_extents.slice_away(sliced_dim).product();
162  if (interface_vars->number_of_grid_points() != interface_grid_points) {
163  *interface_vars =
164  Variables<tmpl::list<TagsToSlice...>>(interface_grid_points);
165  }
166  for (SliceIterator si(element_extents, sliced_dim, fixed_index); si; ++si) {
167  const auto lambda = [&si](auto& interface_tensor,
168  const auto& volume_tensor) noexcept {
169  for (decltype(auto) interface_and_volume_tensor_components :
170  boost::combine(interface_tensor, volume_tensor)) {
171  boost::get<0>(
172  interface_and_volume_tensor_components)[si.slice_offset()] =
173  boost::get<1>(
174  interface_and_volume_tensor_components)[si.volume_offset()];
175  }
176  };
177  expand_pack((lambda(get<TagsToSlice>(*interface_vars), tensors),
178  cpp17::void_type{})...);
179  }
180 }

195 /*!
196  * \ingroup DataStructuresGroup
197  * \brief Adds data on a codimension 1 slice to a volume quantity. The
198  * slice has a constant logical coordinate in direction `sliced_dim`,
199  * slicing the volume at `fixed_index` in that dimension. For
200  * example, to add to the lower boundary of `sliced_dim`, pass `0` for
201  * `fixed_index`; to add to the upper boundary, pass
202  * `extents[sliced_dim] - 1`.
203  *
204  * \see data_on_slice
205  */
206 template <std::size_t VolumeDim, typename TagsList>
207 void add_slice_to_data(const gsl::not_null<Variables<TagsList>*> volume_vars,
208  const Variables<TagsList>& vars_on_slice,
209  const Index<VolumeDim>& extents, const size_t sliced_dim,
210  const size_t fixed_index) noexcept {
211  constexpr const size_t number_of_independent_components =
212  Variables<TagsList>::number_of_independent_components;
213  const size_t volume_grid_points = extents.product();
214  const size_t slice_grid_points = extents.slice_away(sliced_dim).product();
215  ASSERT(volume_vars->number_of_grid_points() == volume_grid_points,
216  "volume_vars has wrong number of grid points. Expected "
217  << volume_grid_points << ", got "
218  << volume_vars->number_of_grid_points());
219  ASSERT(vars_on_slice.number_of_grid_points() == slice_grid_points,
220  "vars_on_slice has wrong number of grid points. Expected "
221  << slice_grid_points << ", got "
222  << vars_on_slice.number_of_grid_points());
223  using value_type = typename Variables<TagsList>::value_type;
224  value_type* const volume_data = volume_vars->data();
225  const value_type* const slice_data = vars_on_slice.data();
226  for (SliceIterator si(extents, sliced_dim, fixed_index); si; ++si) {
227  for (size_t i = 0; i < number_of_independent_components; ++i) {
228  // clang-tidy: do not use pointer arithmetic
229  volume_data[si.volume_offset() + i * volume_grid_points] += // NOLINT
230  slice_data[si.slice_offset() + i * slice_grid_points]; // NOLINT
231  }
232  }
233 }

237 template <typename TagsList>
238 void orient_each_component(
239  const gsl::not_null<Variables<TagsList>*> oriented_variables,
240  const Variables<TagsList>& variables,
241  const std::vector<size_t>& oriented_offset) noexcept {
242  using VectorType = typename Variables<TagsList>::vector_type;
243  tmpl::for_each<TagsList>(
244  [&oriented_variables, &variables, &oriented_offset](auto tag) {
245  using Tag = tmpl::type_from<decltype(tag)>;
246  auto& oriented_tensor = get<Tag>(*oriented_variables);
247  const auto& tensor = get<Tag>(variables);
248  for (decltype(auto) oriented_and_tensor_components :
249  boost::combine(oriented_tensor, tensor)) {
250  VectorType& oriented_tensor_component =
251  boost::get<0>(oriented_and_tensor_components);
252  const VectorType& tensor_component =
253  boost::get<1>(oriented_and_tensor_components);
254  for (size_t s = 0; s < tensor_component.size(); ++s) {
255  oriented_tensor_component[oriented_offset[s]] = tensor_component[s];
256  }
257  }
258  });
259 }

283 /// \ingroup DataStructuresGroup
284 /// Orient variables to the data-storage order of a neighbor element with
285 /// the given orientation.
286 /// @{
287 template <size_t VolumeDim, typename TagsList>
288 Variables<TagsList> orient_variables(
289  const Variables<TagsList>& variables, const Index<VolumeDim>& extents,
290  const OrientationMap<VolumeDim>& orientation_of_neighbor) noexcept {
291  // Skip work (aside from a copy) if neighbor is aligned
292  if (orientation_of_neighbor.is_aligned()) {
293  return variables;
294  }
295 
296  const size_t number_of_grid_points = extents.product();
297  ASSERT(variables.number_of_grid_points() == number_of_grid_points,
298  "Inconsistent `variables` and `extents`:\n"
299  " variables.number_of_grid_points() = "
300  << variables.number_of_grid_points()
301  << "\n"
302  " extents.product() = "
303  << extents.product());
304  Variables<TagsList> oriented_variables(number_of_grid_points);
305  const auto oriented_offset =
306  OrientVariables_detail::oriented_offset(extents, orientation_of_neighbor);
307  OrientVariables_detail::orient_each_component(
308  make_not_null(&oriented_variables), variables, oriented_offset);
309 
310  return oriented_variables;
311 }
