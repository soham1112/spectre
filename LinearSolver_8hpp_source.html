26 /*!
27  * \brief Initializes the DataBox tag related to the linear solver
28  *
29  * Uses the analytic solution to compute the linear solver source.
30  * Assumes the system fields are initialized to zero (see
31  * `Elliptic::Initialization::System`) so we don't have to apply the operator to
32  * the initial guess.
33  *
34  * With:
35  * - `sources_tag` = `db::add_tag_prefix<Tags::Source, system::fields_tag>`
36  *
37  * Uses:
38  * - Metavariables:
39  * - `linear_solver`
40  * - System:
41  * - `fields_tag`
42  * - DataBox:
43  * - `sources_tag`
44  *
45  * DataBox:
46  * - Adds:
47  * - All items in `linear_solver::tags`
48  */
49 template <typename Metavariables>
50 struct LinearSolver {
51  using linear_solver_tags = typename Metavariables::linear_solver::tags;
52  using system = typename Metavariables::system;
53 
54  using sources_tag =
55  db::add_tag_prefix<Tags::Source, typename system::fields_tag>;
56  using fields_operator_tag =
57  db::add_tag_prefix<::LinearSolver::Tags::OperatorAppliedTo,
58  typename system::fields_tag>;
59 
60  using simple_tags = typename linear_solver_tags::simple_tags;
61  using compute_tags = typename linear_solver_tags::compute_tags;
62 
63  template <typename TagsList, typename ArrayIndex, typename ParallelComponent>
64  static auto initialize(
65  db::DataBox<TagsList>&& box,
66  const Parallel::ConstGlobalCache<Metavariables>& cache,
67  const ArrayIndex& array_index,
68  const ParallelComponent* const parallel_component_meta) noexcept {
69  const auto& sources = get<sources_tag>(box);
70 
71  // Starting with x_0 = 0 initial guess, so Ax_0=0
72  auto fields_operator =
73  make_with_value<db::item_type<fields_operator_tag>>(sources, 0.);
74 
75  return linear_solver_tags::initialize(std::move(box), cache, array_index,
76  parallel_component_meta, sources,
77  std::move(fields_operator));
78  }
79 };
80 } // namespace Initialization
81 } // namespace Elliptic
