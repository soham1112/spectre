26 /*!
27  * \brief Computes the sources of the elliptic equations and adds them to the
28  * DataBox
29  *
30  * \note Currently the sources are always retrieved from an analytic solution.
31  *
32  * With:
33  * - `sources_tag` = `db::add_tag_prefix<Tags::Source, system::fields_tag>`
34  *
35  * Uses:
36  * - Metavariables:
37  * - `analytic_solution_tag`
38  * - ConstGlobalCache:
39  * - `analytic_solution_tag`
40  * - System:
41  * - `volume_dim`
42  * - `fields_tag`
43  * - DataBox:
44  * - `Tags::Mesh<volume_dim>`
45  * - `Tags::Coordinates<volume_dim, Frame::Inertial>`
46  *
47  * DataBox:
48  * - Adds:
49  * - `sources_tag`
50  */
51 template <typename Metavariables>
52 struct Source {
53  using system = typename Metavariables::system;
54 
55  using sources_tag =
56  db::add_tag_prefix<Tags::Source, typename system::fields_tag>;
57 
58  using simple_tags = db::AddSimpleTags<sources_tag>;
59  using compute_tags = db::AddComputeTags<>;
60 
61  template <typename TagsList>
62  static auto initialize(
63  db::DataBox<TagsList>&& box,
64  const Parallel::ConstGlobalCache<Metavariables>& cache) noexcept {
65  const auto& inertial_coords =
66  get<Tags::Coordinates<system::volume_dim, Frame::Inertial>>(box);
67  const auto num_grid_points =
68  get<Tags::Mesh<system::volume_dim>>(box).number_of_grid_points();
69 
70  db::item_type<sources_tag> sources(num_grid_points, 0.);
71  // This actually sets the complete set of tags in the Variables, but there
72  // is no Variables constructor from a TaggedTuple (yet)
73  sources.assign_subset(
74  Parallel::get<typename Metavariables::analytic_solution_tag>(cache)
75  .variables(inertial_coords,
76  db::get_variables_tags_list<sources_tag>{}));
77 
78  return db::create_from<db::RemoveTags<>, simple_tags, compute_tags>(
79  std::move(box), std::move(sources));
80  }
81 };
82 } // namespace Initialization
83 } // namespace Elliptic
